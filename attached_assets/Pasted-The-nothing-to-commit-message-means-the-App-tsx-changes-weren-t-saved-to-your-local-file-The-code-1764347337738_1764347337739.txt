const registerForPushNotifications = async () => {
  if (!Device.isDevice) {
    console.log('Push notifications require a physical device');
    return;
  }
  
  const { status } = await Notifications.requestPermissionsAsync();
  if (status !== 'granted') {
    console.log('Notification permission denied');
    return;
  }
  
  const token = (await Notifications.getExpoPushTokenAsync({
    projectId: Constants.expoConfig?.extra?.eas?.projectId,
  })).data;
  
  console.log('Expo Push Token:', token);
  
  // Register with NotifyHound backend
  try {
    const response = await fetch('https://e418e926-1b2b-448a-abdf-a3bf17167ade-00-38cbm3g4eaphv.riker.replit.dev:8000/api/register-token', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        clientId: 'test-client',
        deviceId: Constants.deviceId || `device-${Date.now()}`,
        token: token
      })
    });
    const result = await response.json();
    console.log('Token registered:', result);
  } catch (error) {
    console.log('Failed to register token:', error);
  }
};
